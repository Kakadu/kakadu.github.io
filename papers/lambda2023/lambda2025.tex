\subsection{Capture avoiding substitution}
\label{sec:CAS}

\begin{frame}{Осуществление подстановки (1/2)}
Корректный пример:
\[
\mathfrak{II} = \lam{x}{x}\lam{x}{x} =
\lam{x}{x}\lam{z}{z} =
\]
\[
= [\lam{z}{z}/x]x
= \lam{z}{z} = \mathfrak{I}
\]\\
\pause

\vspace{2em}
Некорректный пример:
\[
\lam{x}{\lam{y}{xy}}y \neq
\lam{y}{yy}
\]
А нужно делать так
\[
\lam{x}{\lam{y}{xy}}y =
\lam{x}{\lam{t}{xt}}y =
\lam{t}{yt}
\]

\end{frame}

\begin{frame}{Осуществление подстановки (2/2)}
Основная идея (capture avoiding substitution):
\begin{itemize}
  \item Если у нас есть
$\lam{x}{e}E$, то мы заменяем все \emph{свободные} вхождения $x$ на $E$
  \item  Если при замене свободная переменная в $E$ вдруг становится связанной, то мы переименовываем связанную переменную в $e$ перед выполнением подстановки
 \end{itemize}\vspace{1cm}

Пример:
\[
\lam{x}{\lam{y}{x\lam{x}{xy}}}y
\]

В $\lam{y}{x\lam{x}{xy}}$ только первый $x$ может быть заменен.
Но перед подстановкой необходимо переименовать $y$ в теле на новое имя $t$
\[
[y/x]\lam{t}{x\lam{x}{xt}} =\lam{t}{y\lam{x}{xt}}
\]
\end{frame}



\begin{frame}{Язык $\Lambda$-выражений}
В начале нужно выбрать язык $\mathcal{V}$ имен переменных. Варианты:

\begin{itemize}
  \item [\faCheck] Непустая последовательность букв: \texttt{[a-z]([a-z])*}
  \item [\faCheckEmpty] Слова в конечном алфавите $\big{\{}s, I, e\big{\}}$: $sIe$, $sIIe$, $sIIIe\dots$
  \item [\faCheckEmpty] \enquote{Безымянные} представления
\end{itemize}\vspace{1em}

Из алфавита строятся слова (предложения) языка: (\emph{лямбда выражения} или \emph{лямбда термы}).
Алфавитом для $\Lambda$ будет $\big{\{} \text{)}, \text{(}, ., \lambda \big{\}}\cup \mathcal{V}$\\\vspace{1em}

Слова в языке $\Lambda$ строятся по следующим грамматическим правилам
\begin{bnf*}
\bnfprod{expr}
  {\bnfpn{varname} \bnfor
    \bnfts{$\big{(}\lambda$} \bnfpn{varname} . \bnfpn{expr} \bnfts{$\big )$} \bnfor
    \bnfts{$\big{(}$} \bnfpn{expr}\bnfpn{expr} \bnfts{$\big )$}}\\
  \bnfprod{varname}
  {\mathcal{V}}
\end{bnf*}
Т.е. $\lambda$-выражение это или \emph{вхождение переменной}, либо $\lambda$-\emph{абстракция}, либо \emph{применение} (аппликация)
\end{frame}


\begin{frame}{Ещё два эквивалентных определения $\Lambda$}
\begin{definition}[Определение языка лямбд №2 (по индукции)]
$\Lambda_1 \equiv V$ --- $\lambda$-выражения высоты 1.\\
$\Lambda_n \equiv \{ \app{l\ }{r} \mid l,r\in \Lambda_m, m<n\}\ \cup\ \{ \lam{v}{r} \mid v\in V, r\in \Lambda_m, m<n \}, n>1$.\\
$\Lambda_{\phantom{n}} \equiv \bigcup_{i\in \mathbb{N}} \Lambda_i$
\end{definition}

\begin{definition}[Определение языка лямбд №3 (через неподвижную точку)]
Заведём оператор $A$ вот так:
$A(X) \equiv V\cup X\cup \{ \app{l\ }{r} \mid l,r\in X\}\ \cup\  \{ \lam{v}{r} \mid r \in X, v\in V\}$

Неподвижная точка оператора $A$~--- все $\lambda$ выражения.
\end{definition}

\end{frame}
